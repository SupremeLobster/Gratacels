void Solucio::calcularMatriuSumes() { // ALGORISME VORAÃ‡

	for(int k=0; k<mida*mida; k++) {
		int suma = 0;
		int i = k;
		int j = aMax;

		while(i<mida*mida) {
			int m = alcades[j];

			while(m>0 && i<mida*mida) {
				suma += j;
				i++;
				m--;
			}

			j--;
		}

		matSumes.push_back(suma);
	}
}
